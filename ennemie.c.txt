void deplacer_ennemie(ennemie *e)
{

	{
		e->direction=0 ; 
	}
	if(e->position.x<=390)
	{	
		e->direction=1 ; 
	}

	{
		e->position.x-=2 ; 
                e->position.y-=2;
	} 

	else if(e->direction==1)
	{
		e->position.x+=2; 

                e->position.y+=2 ;

	}
